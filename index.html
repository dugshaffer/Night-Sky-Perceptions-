let c = document.getElementById('sky'); let ctx = c.getContext('2d'); c.width = window.innerWidth; c.height = window.innerHeight; let objects = []; for(let i=0;i<50;i++) { objects.push({ x: Math.random() c.width, y: Math.random() c.height, speed: Math.random() 0.5+0.1, color: 'white' }) } let plane = { x: c.width/2, y: c.height/2, speed: 2, color: 'red' } function draw() { ctx.fillStyle = 'black'; ctx.fillRect(0,0,c.width,c.height); for(let o of objects) { ctx.fillStyle = o.color; ctx.fillRect(o.x,o.y,1,1); o.x += o.speed; if(o.x > c.width) o.x = 0; } ctx.fillStyle = plane.color; ctx.fillRect(plane.x,plane.y,3,3); plane.x += plane.speed; if(plane.x > c.width+10) plane.x = -10; requestAnimationFrame(draw); } draw();
